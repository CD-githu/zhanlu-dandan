```typescript
import type { Task, TaskInput } from '@/types/task'

export class TaskService {
  async createTask(input: TaskInput): Promise<Task> {
    const id = nanoid()
    const task: Task = {
      id,

  }
```

## 类型安全工具函数

```typescript
// utils/type-guards.ts
import { type Task, type Schedule, type Workflow } from '@/types'

export function isTask(obj: any): obj is Task {
  return obj &&
    typeof obj.id === 'string' &&
    typeof obj.title === 'string' &&
    typeof obj.type === 'string' &&
    typeof obj.status === 'string'
}

export function isSchedule(obj: any): obj is Schedule {
  return obj &&
    typeof obj.id === 'string' &&
    typeof obj.title === 'string' &&
    obj.startTime instanceof Date
}

export function isWorkflow(obj: any): obj is Workflow {
  return obj &&
    typeof obj.id === 'string' &&
    typeof obj.name === 'string' &&
    Array.isArray(obj.nodes)
}

// 类型断言
export function assertTask(obj: any): asserts obj is Task {
  if (!isTask(obj)) {
    throw new Error('Object is not a Task')
  }
}
```

## 设计原则

1. **类型完整性**：所有数据结构都有对应的 类型定义
2. **可扩展性**：所有接口都可以扩展
3. **向后兼容**：添加新字段时使用可选类型
4. **类型推断**：充分利用 TypeScript 的类型推断
5. **运行时验证**：结合类型守卫进行运行时验证

## 常见问题

### Q1: 如何处理日期类型？

使用 JavaScript 原生 `Date` 对象，但在 Tauri 命令中可能需要转换为 ISO 字符串：

```typescript
// 发送到后端时
const data = {
  ...taskInput,
  dueDate: taskInput.dueDate?.toISOString()
}

// 从后端接收时
const task = {
  ...response,
  dueDate: response.dueDate ? new Date(response.dueDate) : undefined
}
```

### Q2: 如何处理枚举类型？

使用字符串字面量联合类型：

```typescript
type TaskStatus = 'todo' | 'in_progress' | 'paused' | 'done'

const status: TaskStatus = 'todo' // ✓ 正确
const status2: TaskStatus = 'invalid' // ✗ 编译错误
```

### Q3: 如何处理可选嵌套对象？

使用可选链和空值合并：

```typescript
interface Task {
  metadata?: {
    author?: string
    createdAt?: Date
  }
}

function getAuthor(task: Task): string {
  return task.metadata?.author ?? 'Unknown'
}
```
