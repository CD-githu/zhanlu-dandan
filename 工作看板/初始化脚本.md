# 项目初始化脚本

## 概述

本文档提供了在不同操作系统下快速初始化工作看板项目的命令脚本。

## Windows PowerShell 脚本

### 初始化脚本 (init.ps1)

```powershell
# 工作看板项目初始化脚本
# 使用方法: .\init.ps1

Write-Host "========================================" -ForegroundColor Cyan
Write-Host "  工作看板项目初始化脚本" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# 检查 Node.js
Write-Host "检查 Node.js..." -ForegroundColor Yellow
try {
    $nodeVersion = node --version
    Write-Host "✓ Node.js 版本: $nodeVersion" -ForegroundColor Green
} catch {
    Write-Host "✗ 未安装 Node.js，请先安装" -ForegroundColor Red
    exit 1
}

# 检查 Rust
Write-Host "检查 Rust..." -ForegroundColor Yellow
try {
    $rustVersion = rustc --version
    Write-Host "✓ Rust 版本: $rustVersion" -ForegroundColor Green
} catch {
    Write-Host "✗ 未安装 Rust，请先安装" -ForegroundColor Red
    Write-Host "下载地址: https://www.rust-lang.org/tools/install" -ForegroundColor Cyan
    exit 1
}

# 检查 Tauri CLI
Write-Host "检查 Tauri CLI..." -ForegroundColor Yellow
try {
    $tauriVersion = cargo tauri --version
    Write-Host "✓ Tauri CLI: $tauriVersion" -ForegroundColor Green
} catch {
    Write-Host "Installing Tauri CLI..." -ForegroundColor Yellow
    cargo install tauri-cli
}

# 创建项目目录结构
Write-Host ""
Write-Host "创建项目结构..." -ForegroundColor Yellow

$directories = @(
    "src",
    "src/components",
    "src/components/common",
    "src/components/task",
    "src/components/schedule",
    "src/components/workflow",
    "src/components/plan",
    "src/components/collaboration",
    "src/components/settings",
    "src/views",
    "src/store",
    "src/services",
    "src/composables",
    "src/types",
    "src/router",
    "src/utils",
    "src/styles",
    "src/assets/images",
    "src/assets/fonts",
    "src-tauri/src",
    "prisma",
    "prisma/migrations",
    "public"
)

foreach ($dir in $directories) {
    if (-not (Test-Path $dir)) {
        New-Item -ItemType Directory -Path $dir -Force | Out-Null
        Write-Host "  ✓ 创建目录: $dir" -ForegroundColor Green
    }
}

# 创建 package.json
Write-Host ""
Write-Host "创建 package.json..." -ForegroundColor Yellow

$packageJson = @{
    name = "workboard"
    version = "0.1.0"
    type = "module"
    scripts = @{
        dev = "vite"
        build = "vue-tsc && vite build"
        preview = "vite preview"
        tauri = "tauri"
        "tauri:dev" = "tauri dev"
        "tauri:build" = "tauri build"
        "postinstall" = "prisma generate"
    }
    dependencies = @{
        "@element-plus/icons-vue" = "^2.3.1"
        "@prisma/client" = "^5.10.0"
        "@tauri-apps/api" = "^2.0.0"
        "dayjs" = "^1.11.10"
        "element-plus" = "^2.5.6"
        "nanoid" = "^5.0.6"
        "pinia" = "^2.1.7"
        "pinia-plugin-persistedstate" = "^3.2.1"
        "vue" = "^3.4.21"
        "vue-router" = "^4.3.0"
    }
    devDependencies = @{
        "@tauri-apps/cli" = "^2.0.0"
        "@types/node" = "^20.11.24"
        "@vitejs/plugin-vue" = "^5.0.4"
        "@vue/tsconfig" = "^0.5.1"
        "prisma" = "^5.10.0"
        "sass" = "^1.71.1"
        "typescript" = "^5.3.3"
        "vite" = "^5.1.5"
        "vite-plugin-element-plus" = "^0.8.0"
        "vue-tsc" = "^2.0.6"
    }
} | ConvertTo-Json -Depth 10

Set-Content -Path "package.json" -Value $packageJson -Encoding UTF8
Write-Host "  ✓ package.json" -ForegroundColor Green

# 创建 tsconfig.json
Write-Host "创建 tsconfig.json..." -ForegroundColor Yellow

$tsconfigJson = @{
    compilerOptions = @{
        target = "ES2020"
        useDefineForClassFields = $true
        module = "ESNext"
        lib = @("ES2020", "DOM", "DOM.Iterable")
        skipLibCheck = $true
        moduleResolution = "bundler"
        allowImportingTsExtensions = $true
        resolveJsonModule = $true
        isolatedModules = $true
        noEmit = $true
        jsx = "preserve"
        strict = $true
        noUnusedLocals = $true
        noUnusedParameters = $true
        noFallthroughCasesInSwitch = $true
        baseUrl = "."
        paths = @{
            "@/*" = @("src/*")
        }
    }
    include = @("src/**/*.ts", "src/**/*.tsx", "src/**/*.vue")
    references = @(@{ path = "./tsconfig.node.json" })
} | ConvertTo-Json -Depth 10

Set-Content -Path "tsconfig.json" -Value $tsconfigJson -Encoding UTF8
Write-Host "  ✓ tsconfig.json" -ForegroundColor Green

# 创建 tsconfig.node.json
Write-Host "创建 tsconfig.node.json..." -ForegroundColor Yellow

$tsconfigNodeJson = @{
    compilerOptions = @{
        composite = $true
        skipLibCheck = $true
        module = "ESNext"
        moduleResolution = "bundler"
        allowSyntheticDefaultImports = $true
    }
    include = @("vite.config.ts")
} | ConvertTo-Json -Depth 10

Set-Content -Path "tsconfig.node.json" -Value $tsconfigNodeJson -Encoding UTF8
Write-Host "  ✓ tsconfig.node.json" -ForegroundColor Green

# 创建 vite.config.ts
Write-Host "创建 vite.config.ts..." -ForegroundColor Yellow

$viteConfig = @"
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import ElementPlus from 'vite-plugin-element-plus'
import { resolve } from 'path'

export default defineConfig({
  plugins: [
    vue(),
    ElementPlus()
  ],
  resolve: {
    alias: {
      '@': resolve(__dirname, 'src')
    }
  },
  server: {
    port: 5173,
    strictPort: true
  },
  clearScreen: false,
  envPrefix: ['VITE_', 'TAURI_'],
  build: {
    target: process.env.TAURI_PLATFORM == 'windows' ? 'chrome105' : 'safari13',
    minify: 'esbuild'
  }
})
"@

Set-Content -Path "vite.config.ts" -Value $viteConfig -Encoding UTF8
Write-Host "  ✓ vite.config.ts" -ForegroundColor Green

# 创建 index.html
Write-Host "创建 index.html..." -ForegroundColor Yellow

$indexHtml = @"
<!DOCTYPE html>
<html lang="zh-CN">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>工作看板</title>
  </head>
  <body>
    <div id="app"></div>
    <script type="module" src="/src/main.ts"></script>
  </body>
</html>
"@

Set-Content -Path "index.html" -Value $indexHtml -Encoding UTF8
Write-Host "  ✓ index.html" -ForegroundColor Green

# 创建 src/main.ts
Write-Host "创建 src/main.ts..." -ForegroundColor Yellow

$mainTs = @"
import { createApp } from 'vue'
import { createPinia } from 'pinia'
import piniaPersist from 'pinia-plugin-persistedstate'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'
import './styles/main.scss'

import App from './App.vue'
import router from './router'

const app = createApp(App)
const pinia = createPinia()

pinia.use(piniaPersist)

app.use(pinia)
app.use(router)
app.use(ElementPlus)

app.mount('#app')
"@

Set-Content -Path "src/main.ts" -Value $mainTs -Encoding UTF8
Write-Host "  ✓ src/main.ts" -ForegroundColor Green

# 创建 src/App.vue
Write-Host "创建 src/App.vue..." -ForegroundColor Yellow

$appVue = @"
<template>
  <el-container class="app-container">
    <el-header class="app-header">
      <AppHeader />
    </el-header>

    <el-container>
      <el-aside :width="sidebarWidth">
        <AppSidebar />
      </el-aside>

      <el-main>
        <router-view />
      </el-main>
    </el-container>
  </el-container>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { ElContainer, ElHeader, ElAside, ElMain } from 'element-plus'
import AppHeader from '@/components/common/AppHeader.vue'
import AppSidebar from '@/components/common/AppSidebar.vue'
import { useAppStore } from '@/store/app'

const appStore = useAppStore()
const sidebarWidth = computed(() => appStore.sidebarCollapsed ? '60px' : '200px')
</script>

<style scoped lang="scss">
.app-container {
  height: 100vh;
  overflow: hidden;
}

.app-header {
  background: #409eff;
  color: white;
  display: flex;
  align-items: center;
  padding: 0 20px;
}

.el-aside {
  background: #545c64;
  color: white;
  transition: width 0.3s;
}

.el-main {
  background: #f5f5f5;
  padding: 20px;
  overflow-y: auto;
}
</style>
"@

Set-Content -Path "src/App.vue" -Value $appVue -Encoding UTF8
Write-Host "  ✓ src/App.vue" -ForegroundColor Green

# 创建 prisma/schema.prisma
Write-Host "创建 prisma/schema.prisma..." -ForegroundColor Yellow

$schemaPrisma = @"
datasource db {
  provider = "sqlite"
  url      = "file:./workboard.db"
}

generator client {
  provider = "prisma-client-js"
}

model Task {
  id          String   @id @default(unique())
  title       String
  description String?
  type        String   // daily, temp, long
  priority    String   // high, medium, low
  status      String   @default("todo") // todo, in_progress, paused, done
  tags        String   @default("")
  assignee    String?
  dueDate     DateTime?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  completedAt DateTime?
}

model Schedule {
  id          String   @id @default(unique())
  title       String
  description String?
  startTime   DateTime
  endTime     DateTime
  allDay      Boolean  @default(false)
  reminderTime Int?
  recurring   String?
  taskId      String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}
"@

Set-Content -Path "prisma/schema.prisma" -Value $schemaPrisma -Encoding UTF8
Write-Host "  ✓ prisma/schema.prisma" -ForegroundColor Green

# 创建 .gitignore
Write-Host "创建 .gitignore..." -ForegroundColor Yellow

$gitignore = @"
# Dependencies
node_modules/
.pnp
.pnp.js

# Testing
coverage/

# Production
dist/
dist-ssr/
*.local

# Editor directories and files
.vscode/*
!.vscode/extensions.json
.idea
.DS_Store
*.suo
*.ntvs*
*.njsproj
*.sln
*.sw?

# Rust
target/

# Prisma
*.db
*.db-journal

# Environment
.env
.env.local
.env.production.local
.env.development.local
.env.test.local
"@

Set-Content -Path ".gitignore" -Value $gitignore -Encoding UTF8
Write-Host "  ✓ .gitignore" -ForegroundColor Green

# 初始化 Tauri
Write-Host ""
Write-Host "初始化 Tauri..." -ForegroundColor Yellow
npm install @tauri-apps/cli --save-dev
npx tauri init --yes --app-name "工作看板" --window-title "工作看板" --dist-dir "../dist" --dev-url "http://localhost:5173"

Write-Host "✓ Tauri 初始化完成" -ForegroundColor Green

# 静默方式安装依赖
Write-Host ""
Write-Host "安装 Node 依赖..." -ForegroundColor Yellow
Start-Process -NoNewWindow -Wait npm -ArgumentList "install", "--silent"
Write-Host "✓ 依赖安装完成" -ForegroundColor Green

# 初始化 Prisma
Write-Host ""
Write-Host "初始化数据库..." -ForegroundColor Yellow
npx prisma generate
npx prisma migrate dev --name init --skip-generate
Write-Host "✓ 数据库初始化完成" -ForegroundColor Green

Write-Host ""
Write-Host "========================================" -ForegroundColor Cyan
Write-Host "  初始化完成！" -ForegroundColor Green
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""
Write-Host "启动开发环境:" -ForegroundColor Cyan
Write-Host "  npm run tauri:dev" -ForegroundColor Yellow
Write-Host ""
Write-Host "构建生产版本:" -ForegroundColor Cyan
Write-Host "  npm run tauri:build" -ForegroundColor Yellow
Write-Host ""
```

   - package.json - Node 依赖配置
   - tsconfig.json - TypeScript 配置
   - vite.config.ts - Vite 构建配置
   - index.html - HTML 入口
   - prisma/schema.prisma - 数据库模型

   - src/main.ts - 应用入口
   - src/App.vue - 根组件

5. **依赖安装**
   - 安装所有 Node 依赖
   - 初始化 Tauri 项目
   - 初始化 Prisma 数据库

6. **版本控制配置**
   - 创建 .gitignore 文件
