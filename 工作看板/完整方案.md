# 工作看板桌面应用 - 完整方案

## 技术栈

### 前端框架
- **Tauri 2.x** - 轻量级桌面应用框架
- **Vue 3.3+** - 渐进式前端框架
- **TypeScript** - 类型安全
- **Vite** - 构建工具
- **Pinia** - 状态管理
- **Vue Router** - 路由管理

### UI组件库
- **Element Plus** - 成熟的桌面UI组件
- **@vueuse/core** - Vue组合式工具集

### 数据库
- **SQLite** - 本地持久化存储
- **Prisma ORM** - 类型安全的数据库操作

### 工具库
- **dayjs** - 日期处理
- **nanoid** - ID生成
- **hotkeys-js** - 快捷键
- **electron-store** - Tauri配置存储

### 3. 实现方案
```typescript
// services/ai-service.ts
export class AIService {
  async generateSummary(request: AISummaryRequest): Promise<AISummaryResponse> {
    const config = await getAIConfig()
    const prompt = this.buildSummaryPrompt(request)

    const response = await fetch(`${config.apiUrl}/chat/completions`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${config.apiToken}`
      },
      body: JSON.stringify({
        model: config.model || 'default',
        messages: [
          { role: 'system', content: '你是一个工作助手...' },
          { role: 'user', content: prompt }
        ]
      })
    })

    return this.parseResponse(response)
  }

  private buildSummaryPrompt(request: AISummaryRequest): string {
    const tasksSummary = request.tasks.map(task =>
      `- ${task.title} (${task.status}): ${task.description}`
    ).join('\n')

    return `
请根据以下工作内容生成工作总结：

时间范围：${request.period.start.toLocaleDateString()} ~ ${request.period.end.toLocaleDateString()}

任务列表：
${tasksSummary}

请提供：
1. 工作概述
2. 主要成果
3. 遇到的挑战
4. 改进建议
`
  }
}
```

## 数据库设计（Prisma Schema）

```prisma
datasource db {
  provider = "sqlite"
  url      = "file:./workboard.db"
}

generator client {
  provider = "prisma-client-js"
}

model Task {
  id          String   @id @default(unique())
  title       String
  description String?
  type        String   // daily, temp, long
  priority    String   // high, medium, low
  status      String   @default("todo") // todo, in_progress, paused, done
  tags        String   @default("") // JSON array as string
  assignee    String?
  dueDate     DateTime?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  completedAt DateTime?
  parentId    String?
  Subtasks    SubTask[]
  Attachments Attachment[]
}

model SubTask {
  id        String @id @default(unique())
  title     String
  completed Boolean @default(false)
  taskId    String
  Task      Task   @relation(fields: [taskId], references: [id], onDelete: Cascade)
}

model Attachment {
  id        String @id @default(unique())
  name      String
  path      String
  size      Int
  taskId    String
  Task      Task   @relation(fields: [taskId], references: [id], onDelete: Cascade)
}

model Schedule {
  id          String        @id @default(unique())
  title       String
  description String?
  startTime   DateTime
  endTime     DateTime
  allDay      Boolean       @default(false)
  reminderTime Int?
  recurring   String?       // JSON as string
  taskId      String?
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt
}

model Workflow {
  id        String   @id @default(unique())
  name      String
  description String?
  nodes     String   // JSON as string
  edges     String   // JSON as string
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Plan {
  id         String     @id @default(unique())
  type       String     // weekly, monthly, yearly
  period     String
  goals      String     // JSON as string
  summary    String?
  aiSummary  String?
  createdAt  DateTime   @default(now())
  updatedAt  DateTime   @updatedAt
}

model User {
  id          String   @id @default(unique())
  name        String
  role        String   // primary, backup, guest
  permissions String   // JSON as string
  lastActive  DateTime @default(now())
  HandoversFrom Handover[] @relation("HandoverFrom")
  HandoversTo   Handover[] @relation("HandoverTo")
  OperationLogs OperationLog[]
}

model Handover {
  id         String   @id @default(unique())
  fromUserId String
  toUserId   String
  note       String
  tasks      String   // JSON array as string
  createdAt  DateTime @default(now())
  acknowledgedAt DateTime?
  FromUser   User     @relation("HandoverFrom", fields: [fromUserId], references: [id])
  ToUser     User     @relation("HandoverTo", fields: [toUserId], references: [id])
}

model OperationLog {
  id        String   @id @default(unique())
  userId    String
  action    String
  details   String   // JSON as string
  timestamp DateTime @default(now())
  User      User     @relation(fields: [userId], references: [id])
}
```
